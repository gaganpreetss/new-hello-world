function getDropzoneFiles() {
	var files = Dropzone.instances[0].files;
	var fileNames = [];
	for( var i in files ) {
		var f = files[i];
		if ( f.accepted && f.xhr && f.xhr.responseText ) {
			var str = f.xhr.responseText;
			var arr = str.split( '||' );
			for( var i = 0; i < arr.length; i++ ) {
				var arr2 = arr[i].split( '|:|' );
				if( arr2.length == 2 && arr2[0] == 'filename' ) fileNames.push( arr2[1] );
			}
		}
	}
	return fileNames;
}
